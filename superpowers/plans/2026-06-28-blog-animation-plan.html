博客动画增强 — 实现计划 ​

For agentic workers: REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (- [ ]) syntax for tracking.

Goal: 为 VitePress 博客全站添加 GSAP 动画效果（滚动渐现、Matrix 雨、打字机、页面过渡、进度条、CSS 动效）。

Architecture: GSAP + ScrollTrigger 作为动画引擎，通过 Vue 自定义指令 v-reveal 和 composables 暴露给 Markdown 页面。MatrixRain / ScrollProgress 作为独立 Vue 组件按需引入。

Tech Stack: VitePress 1.6.4, Vue 3.5, GSAP 3.12, TypeScript

Global Constraints ​

  • gsap ^3.12.0 — 按需引入 ScrollTrigger、TextPlugin
  • 所有 ScrollTrigger 在 onUnmounted 时 kill()
  • SSR 安全：GSAP 调用前检查 typeof window === 'undefined'
  • prefers-reduced-motion: reduce 时跳过所有动画
  • Matrix 雨在视口不可见时暂停（IntersectionObserver）
  • 移动端 Matrix 密度减半

import ScrollTrigger from 'gsap/ScrollTrigger'

gsap.registerPlugin(ScrollTrigger)

/** 元素渐现的默认 from 参数 */
export const revealDefaults: gsap.TweenVars = {
  opacity: 0,
  y: 30,
  duration: 0.5,
  ease: 'power2.out',
}

/** ScrollTrigger 默认配置 */
export const scrollTriggerDefaults: ScrollTrigger.Vars = {
  start: 'top 85%',
  toggleActions: 'play none none none',
}

/**
 * 对元素应用滚动触发渐现动画
 * 返回 tween 实例，调用方负责在 unmount 时 kill()
 */
export function revealElement(el: HTMLElement, overrides?: gsap.TweenVars): gsap.core.Tween {
  return gsap.from(el, {
    ...revealDefaults,
    ...overrides,
    scrollTrigger: {
      trigger: el,
      ...scrollTriggerDefaults,
      ...(overrides?.scrollTrigger as ScrollTrigger.Vars || {}),
    },
  })
}

import { onMounted, onUnmounted, type Ref } from 'vue'
import gsap from 'gsap'
import ScrollTrigger from 'gsap/ScrollTrigger'
import { revealDefaults, scrollTriggerDefaults } from '../animations/presets'

gsap.registerPlugin(ScrollTrigger)

/**
 * 对模板引用元素应用滚动触发的渐现动画。
 * 在组件 onMounted 时创建 ScrollTrigger，onUnmounted 时自动清理。
 *
 * @example
 * const box = ref<HTMLElement | null>(null)
 * useScrollReveal(box, { y: 50 })
 */
export function useScrollReveal(
  target: Ref<HTMLElement | null>,
  options?: gsap.TweenVars,
): void {
  let ctx: gsap.Context | null = null

  onMounted(() => {
    if (typeof window === 'undefined') return
    if (!target.value) return

    const prefersReduced = window.matchMedia('(prefers-reduced-motion: reduce)').matches
    if (prefersReduced) return

    ctx = gsap.context(() => {
      gsap.from(target.value!, {
        ...revealDefaults,
        ...options,
        scrollTrigger: {
          trigger: target.value!,
          ...scrollTriggerDefaults,
          ...(options?.scrollTrigger as Record<string, unknown> || {}),
        },
      })
    })
  })

  onUnmounted(() => {
    ctx?.revert()
  })
}

import type { Directive } from 'vue'
import gsap from 'gsap'
import ScrollTrigger from 'gsap/ScrollTrigger'
import { revealDefaults, scrollTriggerDefaults } from '../animations/presets'

gsap.registerPlugin(ScrollTrigger)

const contexts = new WeakMap<HTMLElement, gsap.Context>()

/**
 * 自定义指令 v-reveal：元素进入视口时从下方渐现。
 *
 * @usage
 * <h2 v-reveal>标题</h2>
 * <div v-reveal="{ y: 20, duration: 0.8 }">自定义参数</div>
 */
export const vReveal: Directive<HTMLElement, gsap.TweenVars | undefined> = {
  mounted(el, binding) {
    if (typeof window === 'undefined') return

    const prefersReduced = window.matchMedia('(prefers-reduced-motion: reduce)').matches
    if (prefersReduced) return

    const overrides = binding.value || {}

    const ctx = gsap.context(() => {
      gsap.from(el, {
        ...revealDefaults,
        ...overrides,
        scrollTrigger: {
          trigger: el,
          ...scrollTriggerDefaults,
          ...(overrides.scrollTrigger as Record<string, unknown> || {}),
        },
      })
    })

    contexts.set(el, ctx)
  },

  unmounted(el) {
    const ctx = contexts.get(el)
    ctx?.revert()
    contexts.delete(el)
  },
}

ts
import { ref } from 'vue'
import gsap from 'gsap'

/**
 * GSAP 驱动的打字机效果 composable。
 * 替代旧版 setInterval 实现，提供自然的随机延迟和光标平滑闪烁。
 *
 * @example
 * const { displayText, cursorVisible, start } = useTypewriter('hello', {
 *   onComplete: () => console.log('done'),
 * })
 * start()
 */
export function useTypewriter(
  text: string,
  options?: {
    /** 每字符基础速度（秒），默认 0.07 */
    speed?: number
    /** 打字完成后回调 */
    onComplete?: () => void
  },
) {
  const displayText = ref('')
  const cursorVisible = ref(true)
  const isComplete = ref(false)

  let cursorTween: gsap.core.Tween | null = null
  let mainTl: gsap.core.Timeline | null = null

  function start() {
    if (typeof window === 'undefined') {
      // SSR fallback：直接显示完整文本
      displayText.value = text
      isComplete.value = true
      return
    }

    const speed = options?.speed ?? 0.07

    mainTl = gsap.timeline({
      onComplete: () => {
        isComplete.value = true
        options?.onComplete?.()

        // 光标平滑闪烁
        cursorTween = gsap.to({ v: 1 }, {
          v: 0,
          duration: 0.55,
          repeat: -1,
          yoyo: true,
          ease: 'power2.inOut',
          onUpdate() {
            cursorVisible.value = !!(this.targets()[0] as { v: number }).v
          },
        })
      },
    })

    // 逐字打印，间隔带随机微延迟
    let elapsed = 0
    for (let i = 0; i < text.length; i++) {
      const jitter = (Math.random() - 0.5) * 0.04 // ±0.02s 随机
      elapsed += speed + jitter
      mainTl.call(
        () => {
          displayText.value = text.slice(0, i + 1)
        },
        undefined,
        elapsed,
      )
    }
  }

  function stop() {
    mainTl?.kill()
    cursorTween?.kill()
  }

  return { displayText, cursorVisible, isComplete, start, stop }
}

<template>
  <canvas ref="canvasRef" class="matrix-rain" aria-hidden="true"></canvas>

<script setup lang="ts">
import { ref, onMounted, onUnmounted } from 'vue'

const canvasRef = ref<HTMLCanvasElement | null>(null)

// Style B：键盘符号 + 十六进制字符，低密度微光
const CHARS = '{}[]()/<>$*#@.0123456789abcdef'
const FONT_SIZE = 14
const BASE_ALPHA = 0.06

interface Drop {
  y: number
  speed: number
  brightness: number
  pauseTimer: number
  pauseDuration: number
}

let drops: Drop[] = []
let canvas: HTMLCanvasElement
let ctx: CanvasRenderingContext2D
let animId = 0
let observer: IntersectionObserver | null = null

function resize() {
  canvas.width = window.innerWidth
  canvas.height = window.innerHeight

  const isMobile = window.innerWidth < 768
  const colWidth = FONT_SIZE * (isMobile ? 3 : 1.8) // 移动端列更稀疏
  const colCount = Math.floor(canvas.width / colWidth)
  drops = Array.from({ length: colCount }, () => ({
    y: Math.random() * canvas.height * -1,
    speed: 0.3 + Math.random() * 0.5,
    brightness: BASE_ALPHA + Math.random() * 0.04,
    pauseTimer: 0,
    pauseDuration: 30 + Math.random() * 100,
  }))
}

function draw() {
  ctx.clearRect(0, 0, canvas.width, canvas.height)
  ctx.font = `${FONT_SIZE}px "JetBrains Mono", "Fira Code", monospace`

  const colW = canvas.width / drops.length

  for (let i = 0; i < drops.length; i++) {
    const d = drops[i]

    // 偶尔停顿 —— 增加"幽灵字符"感
    if (d.pauseTimer > 0) {
      d.pauseTimer--
    } else {
      d.y += d.speed
      if (Math.random() < 0.003) d.pauseTimer = d.pauseDuration
    }

    // 绘制尾部拖尾
    const trailLen = 5 + Math.floor(Math.random() * 8)
    for (let j = 0; j < trailLen; j++) {
      const cy = d.y - j * FONT_SIZE
      if (cy < 0 || cy > canvas.height) continue

      const isHead = j === 0
      const alpha = isHead
        ? d.brightness * 2.5
        : d.brightness * (1 - j / trailLen) * (0.5 + Math.random() * 0.5)

      ctx.fillStyle = `rgba(126, 184, 126, ${alpha.toFixed(3)})`
      ctx.fillText(
        CHARS[Math.floor(Math.random() * CHARS.length)],
        i * colW + 4,
        cy,
      )
    }

    // 落出屏幕后重置到顶部
    if (d.y > canvas.height + trailLen * FONT_SIZE) {
      d.y = -FONT_SIZE * (5 + Math.random() * 15)
      d.speed = 0.3 + Math.random() * 0.5
    }
  }

  animId = requestAnimationFrame(draw)
}

function init() {
  canvas = canvasRef.value!
  ctx = canvas.getContext('2d')!
  resize()
}

onMounted(() => {
  if (typeof window === 'undefined') return
  init()
  draw()

  observer = new IntersectionObserver(
    ([entry]) => {
      if (entry.isIntersecting && !animId) {
        draw()
      } else if (!entry.isIntersecting && animId) {
        cancelAnimationFrame(animId)
        animId = 0
      }
    },
    { threshold: 0 },
  )
  observer.observe(canvas)
  window.addEventListener('resize', resize)
})

onUnmounted(() => {
  if (animId) cancelAnimationFrame(animId)
  observer?.disconnect()
  window.removeEventListener('resize', resize)
})
</script>

  z-index: auto;
  pointer-events: none;

import { nextTick } from 'vue'
import type { Router } from 'vue-router'
import gsap from 'gsap'

/**
 * 注册全局页面切换效果：
 * 1. 路由切换时内容区淡入 + 微上移
 * 2. 滚动时导航栏浮现阴影
 */
export function usePageTransition(router: Router): void {
  if (typeof window === 'undefined') return

  // --- 页面切换过渡 ---
  let isFirstNav = true

  router.afterEach(() => {
    if (isFirstNav) {
      isFirstNav = false
      return
    }

    nextTick(() => {
      const doc = document.querySelector('.VPDoc') as HTMLElement | null
      if (doc) {
        gsap.fromTo(
          doc,
          { opacity: 0, y: 15 },
          { opacity: 1, y: 0, duration: 0.25, ease: 'power2.out' },
        )
      }
    })
  })

  // --- 导航栏滚动阴影 ---
  const navSelector = '.VPNavBar'
  let rafId = 0

  const onScroll = () => {
    if (rafId) return
    rafId = requestAnimationFrame(() => {
      const nav = document.querySelector(navSelector)
      if (nav) {
        nav.classList.toggle('nav-scrolled', window.scrollY > 20)
      }
      rafId = 0
    })
  }

  window.addEventListener('scroll', onScroll, { passive: true })
}
